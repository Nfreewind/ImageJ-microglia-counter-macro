function lower_threshold(percent_threshold){

	h = getHeight(); 
	w = getWidth(); 
	n = w*h; 
	nBins = 256; 
	histMin = 0; 
	histMax = 255; 
	getHistogram(values, counts, nBins); 
	cum = newArray(nBins); 
	cum[0] = counts[0]; 
	cdf = newArray(nBins); 
	cdf[0] = cum[0]/n; 
	fBin = 0;
	for (i = (histMin + 1); i < (histMax + 1); i++){ 
        	cum[i] = counts[i] + cum[i-1]; 
        	cdf[i] = cum[i]/n; 
        	percent = cdf[i]*100;
        	if ((100-percent)>=percent_threshold){
				fBin = i;
			}
	} 
	return fBin;
}
